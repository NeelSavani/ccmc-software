95  inline void addPoint(DistanceType dist, IndexType index)
96  {
97  CountType i;
98  for (i=count; i>0; --i) {
99 #ifdef NANOFLANN_FIRST_MATCH // If defined and two poins have the same distance, the one with the lowest-index will be returned first.
100  if ( (dists[i-1]>dist) || ((dist==dists[i-1])&&(indices[i-1]>index)) ) {
101 #else
102  if (dists[i-1]>dist) {
103 #endif
104  if (i<capacity) {
105  dists[i] = dists[i-1];
106  indices[i] = indices[i-1];
107  }
108  }
109  else break;
110  }
111  if (i<capacity) {
112  dists[i] = dist;
113  indices[i] = index;
114  }
115  if (count<capacity) count++;
116  }

259  {
260  DistanceType result = DistanceType();

262  const T* lastgroup = last - 3;
263  size_t d = 0;
264 
265  /* Process 4 items with each loop for efficiency. */
266  while (a < lastgroup) {
267  const DistanceType diff0 = nanoflann::abs(a[0] - data_source.kdtree_get_pt(b_idx,d++));

272  a += 4;
273  if ((worst_dist>0)&&(result>worst_dist)) {
274  return result;

276  }

278  while (a < last) {
279  result += nanoflann::abs( *a++ - data_source.kdtree_get_pt(b_idx,d++) );
280  }
281  return result;
282  }

457  class PooledAllocator
458  {
459  /* We maintain memory alignment to word boundaries by requiring that all
460  allocations be in multiples of the machine wordsize. */
461  /* Size of machine word in bytes. Must be power of 2. */
462  /* Minimum number of bytes requested at a time from the system. Must be multiple of WORDSIZE. */
463 
464 
465  size_t remaining; /* Number of bytes left in current block of storage. */
466  void* base; /* Pointer to base of current block of storage. */
467  void* loc; /* Current location in block to next allocate memory. */
468  size_t blocksize;
469 
470 
471  public:
472  size_t usedMemory;
473  size_t wastedMemory;
474 
478  PooledAllocator(const size_t blocksize = BLOCKSIZE)
479  {
480  this->blocksize = blocksize;
481  remaining = 0;
482  base = NULL;
483 
484  usedMemory = 0;
485  wastedMemory = 0;
486  }

491  ~PooledAllocator()

504  void* malloc(const size_t req_size)
505  {
506  /* Round size up to a multiple of wordsize. The following expression
507  only works for WORDSIZE that is a power of 2, by masking last bits of
508  incremented size to zero.
509  */
510  const size_t size = (req_size + (WORDSIZE - 1)) & ~(WORDSIZE - 1);
511 
512  /* Check whether a new block must be allocated. Note that the first word
513  of a block is reserved for a pointer to the previous block.
514  */
515  if (size > remaining) {
516 
517  wastedMemory += remaining;
518 
519  /* Allocate new storage. */
520  const size_t blocksize = (size + sizeof(void*) + (WORDSIZE-1) > BLOCKSIZE) ?
521  size + sizeof(void*) + (WORDSIZE-1) : BLOCKSIZE;
522 
523  // use the standard C malloc to allocate memory
524  void* m = ::malloc(blocksize);
525  if (!m) {
526  fprintf(stderr,"Failed to allocate memory.\n");
527  return NULL;
528  }
529 
530  /* Fill first word of new block with pointer to previous block. */
531  ((void**) m)[0] = base;
532  base = m;
533 
534  size_t shift = 0;
535  //int size_t = (WORDSIZE - ( (((size_t)m) + sizeof(void*)) & (WORDSIZE-1))) & (WORDSIZE-1);
536 
537  remaining = blocksize - sizeof(void*) - shift;
538  loc = ((char*)m + sizeof(void*) + shift);
539  }
540  void* rloc = loc;
541  loc = (char*)loc + size;
542  remaining -= size;
543 
544  usedMemory += size;
545 
546  return rloc;
547  }

915  NodePtr divideTree(const IndexType left, const IndexType right, BoundingBox& bbox)
916  {
917  NodePtr node = pool.allocate<Node>(); // allocate memory
918 
919  /* If too few exemplars remain, then make this a leaf node. */
920  if ( (right-left) <= m_leaf_max_size) {
921  node->child1 = node->child2 = NULL; /* Mark as leaf node. */
922  node->lr.left = left;
923  node->lr.right = right;
924 
925  // compute bounding-box of leaf points
926  for (int i=0; i<(DIM>0 ? DIM : dim); ++i) {
927  bbox[i].low = dataset_get(vind[left],i);
928  bbox[i].high = dataset_get(vind[left],i);
929  }
930  for (IndexType k=left+1; k<right; ++k) {
931  for (int i=0; i<(DIM>0 ? DIM : dim); ++i) {

977  {

979  ElementType max_span = bbox[0].high-bbox[0].low;
980  cutfeat = 0;

982  for (int i=1; i<(DIM>0 ? DIM : dim); ++i) {

988  }

992  ElementType min_elem, max_elem;
993  computeMinMax(ind, count, cutfeat, min_elem, max_elem);

995  max_span = max_elem - min_elem;
996

998  size_t k = cutfeat;
999  for (size_t i=0; i<(DIM>0 ? DIM : dim); ++i) {
1000  if (i==k) continue;

1004  span = max_elem - min_elem;
1005  if (span>max_span) {
1006  max_span = span;
1007  cutfeat = i;
1008  cutval = (min_elem+max_elem)/2;
1009  }
1010  }
1011  }

1013  planeSplit(ind, count, cutfeat, cutval, lim1, lim2);
1014 
1015  if (lim1>count/2) index = lim1;
1016  else if (lim2<count/2) index = lim2;
1017  else index = count/2;

1024  ElementType max_span = bbox[0].high-bbox[0].low;

1072  void planeSplit(IndexType* ind, const IndexType count, int cutfeat, DistanceType cutval, IndexType& lim1, IndexType& lim2)
1073  {
1074  /* Move vector indices for left subtree to front of list. */
1075  IndexType left = 0;
1076  IndexType right = count-1;
1077  for (;; ) {
1078  while (left<=right && dataset_get(ind[left],cutfeat)<cutval) ++left;
1079  while (right && left<=right && dataset_get(ind[right],cutfeat)>=cutval) --right;
1080  if (left>right || !right) break; // "!right" was added to support unsigned Index types
1081  std::swap(ind[left], ind[right]);
1082  ++left;
1083  --right;
1084  }
1085  /* If either list is empty, it means that all remaining features
1086  * are identical. Split in the middle to maintain a balanced tree.
1087  */
1088  lim1 = left;
1089  right = count-1;
1090  for (;; ) {
1091  while (left<=right && dataset_get(ind[left],cutfeat)<=cutval) ++left;
1092  while (right && left<=right && dataset_get(ind[right],cutfeat)>cutval) --right;
1093  if (left>right || !right) break; // "!right" was added to support unsigned Index types
1094  std::swap(ind[left], ind[right]);
1095  ++left;
1096  --right;
1097  }
1098  lim2 = left;
1099  }

1125  void searchLevel(RESULTSET& result_set, const ElementType* vec, const NodePtr node, DistanceType mindistsq,
1126  std::vector<DistanceType>& dists, const float epsError) const
1127  {
1128  /* If this is a leaf node, then do check and return. */
1129  if ((node->child1 == NULL)&&(node->child2 == NULL)) {
1130  //count_leaf += (node->lr.right-node->lr.left); // Removed since was neither used nor returned to the user.
1131  DistanceType worst_dist = result_set.worstDist();
1132  for (IndexType i=node->lr.left; i<node->lr.right; ++i) {
1133  const IndexType index = vind[i];// reorder... : i;
1134  DistanceType dist = distance(vec, index, (DIM>0 ? DIM : dim));
1135  if (dist<worst_dist) {
1136  result_set.addPoint(dist,vind[i]);
1137  }
1138  }
1139  return;
1140  }
1141 
1142  /* Which child branch should be taken first? */
1143  int idx = node->sub.divfeat;
1144  ElementType val = vec[idx];
1145  DistanceType diff1 = val - node->sub.divlow;
1146  DistanceType diff2 = val - node->sub.divhigh;
1147 
1148  NodePtr bestChild;
1149  NodePtr otherChild;
1150  DistanceType cut_dist;
1151  if ((diff1+diff2)<0) {
1152  bestChild = node->child1;
1153  otherChild = node->child2;
1154  cut_dist = distance.accum_dist(val, node->sub.divhigh, idx);
1155  }
1156  else {
1157  bestChild = node->child2;
1158  otherChild = node->child1;
1159  cut_dist = distance.accum_dist( val, node->sub.divlow, idx);
1160  }
1161 
1162  /* Call recursively to search next level down. */
1163  searchLevel(result_set, vec, bestChild, mindistsq, dists, epsError);
1164 
1165  DistanceType dst = dists[idx];
1166  mindistsq = mindistsq + cut_dist - dst;
1167  dists[idx] = cut_dist;
1168  if (mindistsq*epsError<=result_set.worstDist()) {
1169  searchLevel(result_set, vec, otherChild, mindistsq, dists, epsError);
1170  }
1171  dists[idx] = dst;
1172  }
