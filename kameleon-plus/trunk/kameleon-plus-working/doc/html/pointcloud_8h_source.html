50  // Returns the distance between the vector "p1[0:size-1]" and the data point with index "idx_p2" stored in the class:
51  inline T kdtree_distance(const T *p1, const size_t idx_p2,size_t size) const
52  {
53  const T d0=p1[0]-pts[idx_p2].x;
54  const T d1=p1[1]-pts[idx_p2].y;
55  const T d2=p1[2]-pts[idx_p2].z;
56  return d0*d0+d1*d1+d2*d2;
57  }
58 
59  // Returns the dim'th component of the idx'th point in the class:
60  // Since this is inlined and the "dim" argument is typically an immediate value, the
61  // "if/else's" are actually solved at compile time.
62  inline T kdtree_get_pt(const size_t idx, int dim) const
63  {
64  if (dim==0) return pts[idx].x;
65  else if (dim==1) return pts[idx].y;
66  else return pts[idx].z;
67  }
68 
69  // Optional bounding-box computation: return false to default to a standard bbox computation loop.
70  // Return true if the BBOX was already computed by the class and returned in "bb" so it can be avoided to redo it again.
71  // Look at bb.size() to find out the expected dimensionality (e.g. 2 or 3 for point clouds)
72  template <class BBOX>
73  bool kdtree_get_bbox(BBOX &bb) const { return false; }
