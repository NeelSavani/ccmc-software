220  * every position is given a weight in [0,1]. The weights (lambdas) will be linearly precise:
221  * Sum_i lambda_i position[i] = point
222  *
223  * See Langer et. al, Spherical Barycentric Coordinates, 2006 Eurographics Symposium on Geometry
224  */
225 
226 
227  // reset lambda, lambdatemp in case this method was previously called
228  for (int i = 0; i < lambda.size(); i++){
229  lambda[i]=0;
230  lambdatemp[i]= 0;
231  }
232

235  facevectors[i] = vZero;
236  }
237 
238

240  * Set distance vectors to point
241  * if |vi-x|==0, return lambda_i=1, lambda_(j!=i)=0
242  * else, store normalized distance vectors U_i
243  */

246  distances[i] = vZero;
247  distances[i] += positions[i];
248  distances[i] -= point;
249  if (distances[i].length() == 0){

251  thetaMax = piOver2;
252  end = clock(); interpolationTime += end-start;
253  return;
254  }

265  for (int j = faces[i]; j < faces[i+1]; j++){

268  if (u1xu2.length() == 0){

273  }

276  u1xu2.norm();

279  }

283  stage1 += t1-t0;
284 
285

291  thetaFace = 0;
292  for (int j = faces[i]; j < faces[i+1]; j++){
293  vertex = loops[j];
294  Vector<T>::angle(theta[j], facevectors[i],distanceNorms[vertex]);
295  thetaFace += theta[j];
296  }
297  thetaFace /= faces[i+1] - faces[i];
298  if (thetaFace < thetaMin){
299  thetaMin = thetaFace;
300  }
301  if (thetaFace > thetaMax){ //want thetaMax to be the largest angle
302  thetaMax = thetaFace;
303  closestFace = i;
304  }
305  }
306

308  const T infinity = numeric_limits<T>::infinity();
309 
310  // true if theta > pi/2 or theta < pi/2
311  if (!(-infinity < tan(thetaMax) && infinity > tan(thetaMax))){ //point is in same plane as the cell face
312  setFloaterCoordinates(closestFace,point);
313  this->isInside = true;
314  end = clock(); interpolationTime += end-start;
315  return;
316  }
317 
318  if (tan(thetaMax) < T(0)){ // theta > pi/2, point is outside polyhedron
319  this->isInside = false;
320  end = clock(); interpolationTime += end-start;
321  return; //lambdas stay at 0
322  }
323 
324  t2 = clock();
325  stage2 += t2-t1;
326 
327  //Else point is inside the polyhedron, update lambdas
328  this->isInside = true;
329 
330  T alpha, alphaMinus, tangentSum;
331  T sum=T(0);
332 // Vector<T> ufxu0, ufxu1, ufxu2, uf;
333 
334  for (int i = 0; i < facevectors.size();i++){
335  uf = facevectors[i];
336  uf.norm();
337  for (int j = faces[i]; j< faces[i+1]; j++){
338  vertexMinus = loops[minus[j]];
339  vertex = loops[j];
340  vertexPlus = loops[plus[j]];
341

343  Vector<T>::cross(ufxu1,uf,distanceNorms[vertex]);
344  Vector<T>::cross(ufxu2,uf,distanceNorms[vertexPlus]);
345 
346  Vector<T>::angle(alpha, ufxu1,ufxu2);
347  Vector<T>::angle(alphaMinus, ufxu0,ufxu1);
348

350 
351  lambdatemp[vertex] = (facevectors[i].length()/distances[vertex].length())*tangentSum/T(sin(theta[j]));
352  sum = sum + tangentSum/T(tan(theta[j]));
353  }
354

356  vertex = loops[j];

360 
361  }

364  * Normalize weights
365  */

372  }

411  T weight, weightSum = 0;
412

414  vertexMinus = loops[minus[i]];
415  vertex = loops[i];
416  vertexPlus = loops[plus[i]];
417 
418  weight = tan(alpha[minus[i]]/T(2))+tan(alpha[i]/T(2));
419  weight = weight/distances[vertex].length();
